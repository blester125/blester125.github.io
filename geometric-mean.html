\[ \newcommand{\gm}{\mathop{\rm gm}\nolimits} \]

\[ \begin{align} \gm &= \sqrt[n]{\prod_i^n x_i} \\ \gm &= (\prod_i^n x_i)^{n^{-1}} \\ e^{\log \gm} &= e^{\log ((\prod_i^n x_i)^{n^{-1}})} \\ \gm &= e^{\log ((\prod_i^n x_i)^{n^{-1}})} \\ \gm &= e^{n^{-1} \log (\prod_i^n x_i)} \\ \gm &= e^{n^{-1} \sum_i^n \log x_i} \\ \gm &= e^{\frac{1}{n} \sum_i^n \log x_i} \\ \gm &= e^{\text{mean of } \log x} \\ \end{align} \]
